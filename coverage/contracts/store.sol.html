pragma solidity >=0.7.0 <0.9.0;
 
contract Store {
    string private name;
 
    constructor(string memory _name) {
        name = _name;
    }
 
    // setter method - costs gas
    function changeName(string memory _name) public {
        name = _name;
    }
 
    // getter method
    function getName() public view returns (string memory) {
        return name;
    }
}
